5. Write a function to find the sum of all integers between any two values

int_sum <- function(x, y) {
  temp = 0
  jeff_sum = 0
  if (x > y){
    temp = x
    x = y
    y = temp
  }
  for (i in x:y){
    jeff_sum = jeff_sum + i
  }
  return(jeff_sum)
  print(jeff_sum)
}

  prime = TRUE
  for (j in 2:(i-1)) {
    if (i %% j == 0) {
      prime = FALSE
    }

  if (prime == TRUE) {

10a. Write your own functions to give descriptive

statistics for a vector variable: Mean

jeff_mean <- function(x) {
  jeff_sum <- 0
  for (i in 1:(length(x))) {
    jeff_sum <- jeff_sum + (x[i]/length(x))
  }
  return(jeff_sum)
  print(jeff_sum)
}
